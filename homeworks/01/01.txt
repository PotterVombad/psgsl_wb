#### Создал докер контейнер с 17 постгресом и зашел в него 
docker run --name postgres17 \
  -e POSTGRES_PASSWORD=123 \
  -p 5432:5432 \
  -d postgres:17

#### Скачала, распаковал и закинул бд в постгрес 

wget https://storage.googleapis.com/thaibus/thai_small.tar.gz 

tar -xf thai_small.tar.gz

docker exec -i postgres17 psql -U postgres < thai.sql

#### Зашел в постгрес 

docker exec -it postgres17 psql -U postgres
